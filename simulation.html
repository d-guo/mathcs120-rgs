Simulation using Random Graphs

Group

Description

In part I, we simulate the random graphs

  1. G(n,1n1.1)G(n,1n1.1)
  2. G(n,1n)G(n,1n)
  3. G(n,4n)G(n,4n)
  4. G(n,log(n)n)G(n,log(n)n)

for various nn. For each nn, we run mm simulations and plot the number of triangles as a function of nn, as well as the corresponding variance.

In part II, we simulate G(n,n)G(n,n) with =2=2 for various nn and plot the size of the largest component as a function of nn.

# function for simulating a random graph with n nodes and probability p_n
# runs m simulations
def sim_single_RG(n, p_n, m):
    """
    inputs
    n: # of nodes
    p_n: a lambda to calculate probability of an edge as a function of n
    m: # of sims
    
    runs m simulations of the specified random graph
    
    returns tuple (average # of triangles, variance)
    """
    
    num_triangles = []
    
    for _ in range(m):
        if n in [0, 1, 2]:
            return 0, 0

        # initializes an empty graph
        G = nx.Graph()

        # create n nodes
        G.add_nodes_from(range(n))

        # create edges with probability p
        for i in range(n):
            for j in range(i + 1, n):
                if random.uniform(0, 1) < p_n(n):
                    G.add_edge(i, j)

        # count triangles
        # nx.algorithms.cluster.triangles(G1) returns a dictionary formatted as
        # {node_i: number of triangles with node_i as a vertex}
        # to find total triangles, we sum all of these and divide by 3
        num_triangles.append(int(sum(nx.algorithms.cluster.triangles(G).values()) / 3))
    
    return sum(num_triangles) / len(num_triangles), variance(num_triangles)

# function for simulating any type of graph for iterations in [a, b) with step size s
# type of graph determined by p_n lambda probability of edge wrt n
def run_sim(p_n):
    # run simulations for graph and format the data for plotting
    G_num_triangles = []
    G_variance = []

    for n in range(a, b, s):
        avg_num_triangles, var = sim_single_RG(n, p_n, m)
        G_num_triangles.append(avg_num_triangles)
        G_variance.append(var)
    
    return G_num_triangles, G_variance

# function for simulating a random graph with n nodes and probability p_n
# runs m simulations
def sim_single_RG_c(n, p_n, m):
    """
    inputs
    n: # of nodes
    p_n: a lambda to calculate probability of an edge as a function of n
    m: # of sims
    
    runs m simulations of the specified random graph
    
    returns tuple (average size of largest connected component, variance)
    """
    
    largest_c_size = []
    
    for _ in range(m):
        if n in [0, 1, 2]:
            return 0, 0

        # initializes an empty graph
        G = nx.Graph()

        # create n nodes
        G.add_nodes_from(range(n))

        # create edges with probability p
        for i in range(n):
            for j in range(i + 1, n):
                if random.uniform(0, 1) < p_n(n):
                    G.add_edge(i, j)

        # count size of largest connected component
        # max(nx.algorithms.components.connected_components(G), key=len) returns the largest connected component
        largest_c_size.append(len(max(nx.algorithms.components.connected_components(G), key=len)))
    
    return sum(largest_c_size) / len(largest_c_size), variance(largest_c_size)
